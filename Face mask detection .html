# All images will be rescaled by 1./255

                                   rotation_range=40,
                                   width_shift_range=0.2,
                                   height_shift_range=0.2,
                                   shear_range=0.2,
                                   zoom_range=0.2,
                                   horizontal_flip=True,
                                   fill_mode='nearest')
validation_datagen = ImageDataGenerator(rescale=1/255)

# Flow training images in batches of 128 using train_datagen generator
train_generator = train_datagen.flow_from_directory(
        r'C:\Users\Dell\Desktop\facemask\Dataset\train',  # This is the source directory for training images
        target_size=(250, 250),  # All images will be resized to 150x150
        batch_size=128,
        # Since we use binary_crossentropy loss, we need binary labels
        class_mode='binary')

import cv2
from keras.preprocessing import image

# Read the input image
image = cv2.imread('C:\\Users\\Dell\\Desktop\\mo2.jpg')
font = cv2.FONT_HERSHEY_SIMPLEX 




   

if classes[0][0]<=0.5:
    image = cv2.putText(image, 'mask', (50,50), font,  
                   1, (255, 0, 0) , 2, cv2.LINE_AA)
else :
    image = cv2.putText(image, 'no mask', (50,50), font,  
                   1, (255, 0, 0) , 2, cv2.LINE_AA)

gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY) 

# Detect faces
faces = face_cascade.detectMultiScale(gray, 1.1, 4)
# Draw rectangle around the faces
for (x, y, w, h) in faces:
    cv2.rectangle(image, (x, y), (x+w, y+h), (255, 0, 0), 2)
# Display the output
cv2.imshow('face mask detection ', image)
